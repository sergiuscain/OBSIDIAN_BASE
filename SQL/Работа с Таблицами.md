##  Работа с Таблицами
### -- Создание таблицы
```sql
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL
);
```
### -- Добавление столбца
```sql
ALTER TABLE users ADD COLUMN email VARCHAR(100);
```
### -- Удаление таблицы
```sql
DROP TABLE IF EXISTS users;
```

-- СВЯЗИ ТАБЛИЦ:

PRIMARY KEY -- первичный ключ.
PRIMARY KEY(unit_id, lesson_id) -- внешний первичный ключ из двух полей.
FOREIGN KEY (unit_id)  -- внешний ключ unit_id
	REFERENCES units (id)   -- ссылается на таблицу units и поле id в ней.
```sql
CREATE TABLE units(
    id INT PRIMARY KEY,
    course_id INT NOT NULL,
    title TEXT,
    FOREIGN KEY (course_id)
        REFERENCES courses (id)
);

CREATE TABLE lessons(
    id INT PRIMARY KEY,
    title TEXT,
    epic_count INT NOT NULL,
    abuse_count INT NOT NULL
);

CREATE TABLE unit_lessons(   -- таблицв для связи многие ко многим
    unit_id INT,
    lesson_id INT,
    PRIMARY KEY(unit_id, lesson_id),
    FOREIGN KEY (unit_id)
        REFERENCES units (id),
    FOREIGN KEY (lesson_id)
        REFERENCES lessons (id)
);
```
